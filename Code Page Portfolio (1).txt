  <script>
    console.log("Portfolio page loaded");
    const filters = document.querySelectorAll(".filters select");
    filters.forEach(select => select.addEventListener("change", filterArticles));

    function filterArticles() {
      const selected = {
        product: document.getElementById("productType").value.toLowerCase(),
        country: document.getElementById("country").value.toLowerCase(),
        year: document.getElementById("year").value.toLowerCase(),
        sector: document.getElementById("sector").value.toLowerCase(),
      };

      document.querySelectorAll(".article").forEach(article => {
        const matches =
          (!selected.product || article.dataset.product.includes(selected.product)) &&
          (!selected.country || article.dataset.country.toLowerCase().includes(selected.country)) &&
          (!selected.year || article.dataset.year === selected.year) &&
          (!selected.sector || article.dataset.sector.toLowerCase().includes(selected.sector));

        article.style.display = matches ? "block" : "none";
      });
    }
  </script>
